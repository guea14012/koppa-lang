What it looks like

Clean, readable syntax built for security work

port_scan.kop
hash_demo.kop
dns_recon.kop
# Scan a host and print open ports
import scan, log

let target = "192.168.1.1"
let ports  = [22, 80, 443, 8080, 3306]

log.info("Scanning {target}...")

for port in ports {
    if scan.tcp(target, port) {
        log.success("[+] {port} open — {scan.service(port)}")
    }
}

log.info("Done.")
# Hash and encode data
import hash, encode, log

let secret = "password123"

log.info("MD5    : {hash.md5(secret)}")
log.info("SHA256 : {hash.sha256(secret)}")
log.info("NTLM   : {hash.ntlm(secret)}")

let b64 = encode.b64_encode(secret)
log.info("Base64 : {b64}")
log.info("Decode : {encode.b64_decode(b64)}")
# DNS reconnaissance
import dns, log

let targets = ["example.com", "google.com", "github.com"]

for domain in targets {
    let ip  = dns.resolve(domain)
    let rev = dns.reverse(ip)
    log.info("{domain}  ->  {ip}")
    log.info("  reverse: {rev}")
}

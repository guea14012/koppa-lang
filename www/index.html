# Scan a host and print open ports
import scan, log

let target = "192.168.1.1"
let ports  = [22, 80, 443, 8080, 3306]

log.info("Scanning {target}...")

for port in ports {
    if scan.tcp(target, port) {
        log.success("[+] {port} OPEN — {scan.service(port)}")
    }
}

log.info("Done.")
# Hash and encode data
import crypto, log

let secret = "password123"

log.info("MD5    : {crypto.hash_md5(secret)}")
log.info("SHA256 : {crypto.hash_sha256(secret)}")
log.info("NTLM   : {crypto.hash_ntlm(secret)}")

let b64 = crypto.encode_base64(secret)
log.info("Base64 : {b64}")
log.info("Decode : {crypto.decode_base64(b64)}")
# DNS reconnaissance
import recon, log

let targets = ["example.com", "google.com", "github.com"]

for domain in targets {
    let ip  = recon.dns_resolve(domain)
    let rev = recon.dns_reverse(ip)
    log.info("{domain}  ->  {ip}")
    log.info("  reverse: {rev}")
}

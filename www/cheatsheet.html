Result Type
fn safe_scan(host, port) {
    try {
        return Ok(scan.tcp(host, port))
    } catch(e) {
        return Err(e)
    }
}

match safe_scan("t.com", 80) {
    Ok(r)  => log.success(r),
    Err(e) => log.error(e)
}

# Control Flow
If / Elif / Else
if score > 90 {
    log.success("Critical")
} elif score > 50 {
    log.warn("Medium")
} else {
    log.info("Low")
}
For Loop
for port in [22, 80, 443] {
    log.info("Port: {port}")
}

# with condition
for item in list {
    if item == null { continue }
    process(item)
}
While Loop
var i = 0
while i < 10 {
    log.info("i = {i}")
    i = i + 1
}
Match (Pattern)
match service {
    "http"  => http.scan(target),
    "ssh"   => ssh.audit(target),
    "smb"   => smb.enum(target),
    _       => log.info("Unknown: {service}")
}
# Error Handling
Try / Catch
try {
    let data = fs.read("/etc/shadow")
    log.info(data)
} catch(err) {
    log.error("Failed: {err}")
}
Throw
fn divide(a, b) {
    if b == 0 {
        throw "Division by zero"
    }
    return a / b
}
# Pipeline Operator
Chain Operations with |>
target
    |> recon.dns_resolve()
    |> scan.tcp([80, 443])
    |> parse.html_links()
    |> report.save("out.json")
Parallel Block
parallel {
    scan_port("target.com", 80)
    scan_port("target.com", 443)
    scan_port("target.com", 8080)
}
# all three run concurrently

# Full Script Example
Web Recon + Report
import dns, http, parse, fuzz, report, log

fn recon_target(domain) {
    log.info("===== Recon: {domain} =====")

    # DNS
    let ip  = dns.resolve(domain)
    let rev = dns.reverse(ip)
    log.info("IP     : {ip}")
    log.info("Reverse: {rev}")

    # HTTP
    let url  = "https://{domain}"
    let resp = http.get(url)
    log.info("Status : {resp.status}")
    log.info("Title  : {parse.html_title(resp.body)}")

    # Quick fuzz
    let dirs = fuzz.dirs(url, ["admin", "login", "api", "backup"])
    for hit in dirs {
        log.warn("Found: {hit}")
    }

    # Build findings
    let findings = []
    if dirs.len > 0 {
        findings.push(report.finding("Exposed Directory", "Medium", "Found {dirs.len} path(s)"))
    }
    return findings
}

fn main() {
    let findings = recon_target("example.com")

    if findings.len > 0 {
        report.save(findings, "report.html", "html")
        log.success("Report saved.")
    } else {
        log.info("No findings.")
    }
}
